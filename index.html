1. CONTROLLER - HomeController.cs

Controllers/HomeController.cs

using Microsoft.AspNetCore.Mvc;
using System.Diagnostics;
using AllCRUDOperations.Models;

namespace AllCRUDOperations.Controllers
{
    public class HomeController : Controller
    {
        SampleDbContext dbContext = new SampleDbContext();

        public IActionResult Index()
        {
            return View(dbContext.Employees.ToList());
        }

        public IActionResult Create()
        {
            return View();
        }

        public IActionResult AfterCreate(Employee emp)
        {
            dbContext.Add(emp);
            dbContext.SaveChanges();
            return Redirect("/Home/Index");
        }

        public IActionResult Edit(int? id)
        {
            Employee edited = dbContext.Employees.Find(id);
            return View("Edit", edited);
        }

        public IActionResult AfterEdit(Employee emp)
        {
            Employee update = dbContext.Employees.Find(emp.No);
            update.Name = emp.Name;
            update.Address = emp.Address;

            dbContext.SaveChanges();
            return Redirect("/Home/Index");
        }

        public IActionResult delete(int? id)
        {
            Employee deleted = dbContext.Employees.Find(id);
            dbContext.Remove(deleted);
            dbContext.SaveChanges();
            return Redirect("/Home/Index");
        }
    }
}

2d. SampleDbContext.cs

Models/SampleDbContext.cs

using System;
using System.Collections.Generic;
using Microsoft.EntityFrameworkCore;

namespace AllCRUDOperations.Models;

public partial class SampleDbContext : DbContext
{
    public SampleDbContext()
    {
    }

    public SampleDbContext(DbContextOptions<SampleDbContext> options)
        : base(options)
    {
    }

    public virtual DbSet<Employee> Employees { get; set; }

    public virtual DbSet<Subject> Subjects { get; set; }

    public virtual DbSet<Trainer> Trainers { get; set; }

    protected override void OnConfiguring(DbContextOptionsBuilder optionsBuilder)
        => optionsBuilder.UseSqlServer("Data Source=(LocalDB)\\MSSQLLocalDB;Initial Catalog=SampleDB;Integrated Security=True");

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        modelBuilder.Entity<Employee>(entity =>
        {
            entity.HasKey(e => e.No);

            entity.ToTable("Employee");

            entity.Property(e => e.Address).HasMaxLength(50);
            entity.Property(e => e.Name).HasMaxLength(50);
        });

        modelBuilder.Entity<Subject>(entity =>
        {
            entity.ToTable("Subject");

            entity.Property(e => e.SubjectName).HasMaxLength(50);
        });

        modelBuilder.Entity<Trainer>(entity =>
        {
            entity.ToTable("Trainer");

            entity.Property(e => e.TrainerName).HasMaxLength(50);
        });

        OnModelCreatingPartial(modelBuilder);
    }

    partial void OnModelCreatingPartial(ModelBuilder modelBuilder);
}
