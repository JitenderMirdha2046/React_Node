1. EMPLOYEE SALARY SHELL SCRIPT — employee.sh

#!/bin/bash

echo "Enter details for 10 employees:"

for ((i=1; i<=10; i++)); do
    echo "Employee $i name:"
    read names[$i-1]
    echo "Employee $i salary:"
    read salaries[$i-1]
done

total=0
max_sal=0
max_idx=0

for ((i=0; i<10; i++)); do
    echo "${names[$i]} - ${salaries[$i]}"
    total=$((total + ${salaries[$i]}))
    if [ ${salaries[$i]} -gt $max_sal ]; then
        max_sal=${salaries[$i]}
        max_idx=$i
    fi
done

avg=$((total / 10))

echo ""
echo "===== RESULTS ====="
echo "Highest Salary: ${names[$max_idx]} with $max_sal"
echo "Total Salary: $total"
echo "Average Salary: $avg"

echo ""
echo "Sorted by Salary (ascending):"
for ((i=0; i<10; i++)); do
    echo "${salaries[$i]}:${names[$i]}"
done | sort -n | while IFS=: read sal name; do echo "$name - $sal"; done

echo ""
echo "Sorted by Name:"
for ((i=0; i<10; i++)); do
    echo "${names[$i]}:${salaries[$i]}"
done | sort | while IFS=: read name sal; do echo "$name - $sal"; done

3. DOCKERFILE (Multi-stage: OpenJDK + Nginx)

# Stage 1: Build with OpenJDK
FROM openjdk:11 AS builder
WORKDIR /app
COPY employee.sh .
COPY Employee.java .
RUN javac Employee.java
RUN chmod +x employee.sh

# Stage 2: Serve with Nginx
FROM nginx:alpine
WORKDIR /usr/share/nginx/html
COPY --from=builder /app/employee.sh /app/
COPY --from=builder /app/Employee.class /app/
COPY --from=builder /app/Employee.java /app/

# Copy default nginx config
COPY nginx.conf /etc/nginx/nginx.conf

# When container starts, run script then java, then nginx
CMD ["sh", "-c", "cd /app && ./employee.sh && java Employee && nginx -g 'daemon off;'"]

nginx.conf

events {
    worker_connections 1024;
}

http {
    server {
        listen 80;
        server_name localhost;

        location / {
            root /usr/share/nginx/html;
            index index.html;
        }

        location /app {
            alias /app;
        }
    }
}
